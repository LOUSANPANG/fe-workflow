## 配置方面

### 构建 & 加载
- 手动分包
```ts
// 将开发代码文件和公共工具文件分离
// 开发代码经常改动，导致打包文件指纹有变化，用户访问会重新下载文件

export default defineConfig({
  build: {
    rollupOptions: {
      // manualChunks: { vendor: ['lodash', 'vue'] }, // 外部依赖少 可以手动添加
      manualChunks(id) { // 外部依赖多 在执行时添加
        if (id.includes('node_modules')) {
          return 'vendor'
        }
      },
    },
  },
})
```
