#Nginx
```bash
# =========================== 1. 创建 nginx Dockerfile
sudo mkdir -p /usr/local/nginx
touch Dockerfile

# =========================== 2. 编辑 Dockerfile
# 使用官方最新的 Nginx 镜像作为基础镜像
FROM nginx:latest
# 维护者信息
LABEL maintainer="LOUSANPANG<1271255653@qq.com>"
# 设置环境变量，避免一些安装包提示的问题
ENV DEBIAN_FRONTEND=noninteractive
# 更新包索引，安装 vim，同时清理缓存以减少镜像体积
RUN apt-get update && \
    apt-get install -y vim && \
    apt-get clean && \
    rm -rf /var/lib/apt/lists/*
# 创建自定义配置目录（如果需要可以修改路径）
# RUN mkdir -p /etc/nginx/conf.d
# 将工作目录设置为 Nginx 的默认配置目录
WORKDIR /etc/nginx
# 暴露端口
EXPOSE 80
# 将主机配置文件或静态资源挂载到容器（如果需要可以添加）
# COPY ./container-mysql.conf /etc/nginx/nginx.conf
# 设置容器启动时运行的默认命令
CMD ["nginx", "-g", "daemon off;"]

# =========================== 3. 构建镜像
sudo docker build -t container-mysql /usr/local/nginx

# =========================== 4. 运行容器
sudo docker run --name container-mysql -p 80:80 -d container-mysql

# =========================== 5. 进入容器并配置nginx.conf
sudo docker exec -it container-mysql /bin/bash
  vim /etc/nginx/nginx.conf

# =========================== 6. 自动化重载
sudo docker exec container-mysql nginx -s reload

# =========================== 7. 启用、停止、删除
docker start container-mysql
docker stop container-mysql
docker rm -f container-mysql
```

#MySQL
```bash
# =========================== 1. 创建 mysql Dockerfile
sudo mkdir -p /usr/local/docker-mysql
cd /usr/local/docker-mysql
touch Dockerfile

# =========================== 2. 编辑 Dockerfile
# 使用官方 MySQL 镜像
FROM mysql:latest
# 设置环境变量，用于自动配置
ENV MYSQL_ROOT_PASSWORD=abcd1234 \
    MYSQL_DATABASE=my_db \
    MYSQL_USER=lousanpang \
    MYSQL_PASSWORD=abcd1234
# 设置 MySQL 数据存储目录（可选）
VOLUME /var/lib/mysql
# 暴露 MySQL 默认端口
EXPOSE 3306

# =========================== 3. 构建镜像
sudo docker build -t container-mysql /usr/local/docker-mysql

# =========================== 4. 运行容器
docker run -d \
  --name container-mysql \
  -p 3306:3306 \
  -v /mnt/user/mysql-data:/var/lib/mysql \
  container-mysql

# =========================== 5. 进入容器并配置
sudo docker exec -it container-mysql /bin/bash
  mysql -u root -p
  show databases;
  use my_db;
  show tables;
  exit

# =========================== 6. 启用、停止、删除
docker ps
docker start container-mysql
docker stop container-mysql
docker rm -f container-mysql

# =========================== 7. 登录 MySQL 容器
docker exec -it container-mysql mysql -uroot -p
# 查看初始化的数据库和用户
SHOW DATABASES;
SELECT User, Host FROM mysql.user;
```

#docker
```bash
FROM nginx:latest
LABEL maintainer="LOUSANPANG<1271255653@qq.com>"

COPY docker/nginx.conf /etc/nginx/nginx.conf

RUN mkdir -p /root/www/web/dist
COPY dist /root/www/web/dist

EXPOSE 80

CMD ["nginx", "-g", "daemon off;"]
tail -f /var/log/nginx/access_www.log
```

#nginx
```bash
user nginx;  # 设置运行Nginx的用户
worker_processes auto;  # 自动设置工作进程数

events {
    worker_connections 1024;  # 每个进程的最大连接数
}

http {
    include /etc/nginx/mime.types;  # MIME类型配置
    default_type application/octet-stream;  # 默认MIME类型

    sendfile on;  # 启用高效文件传输
    keepalive_timeout 65;  # 保持连接的超时时间

    server {
        listen 80;  # 监听80端口
        server_name localhost;  # 根据需要更改为你的域名或IP

        location / {
            root /root/qiuyan/web;  # 代理到此目录
            index index.html index.htm;  # 默认文件
            try_files $uri $uri/ =404;  # 文件查找策略
        }

        error_page 404 /404.html;  # 404错误页面
        location = /404.html {
            internal;  # 仅内部使用
        }

        error_page 500 502 503 504 /50x.html;  # 500系列错误页面
        location = /50x.html {
            internal;  # 仅内部使用
        }
    }

    # 可添加其他server块或配置
}
```
