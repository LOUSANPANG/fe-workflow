module.exports = {  resolve:  {    extensions: ['.js', '.jsx', '.ts', '.tsx']  },  module: {    rules: [      {        test: /\.(ts|tsx)$/,        use: [          'babel-loader',          'ts-loader'        ],        exclude: /node_modules/      }    ]  }}
